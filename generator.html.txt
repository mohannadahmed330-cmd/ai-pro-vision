    <script type="module">
        const apiKey = ""; 
        let base64Image = null;
        let selectedCount = 2;

        const promptInput = document.getElementById('promptInput');
        const generateBtn = document.getElementById('generateBtn');
        const analyzeBtn = document.getElementById('analyzeBtn');
        const imageUpload = document.getElementById('imageUpload');
        const imgPreview = document.getElementById('imgPreview');
        const loadingOverlay = document.getElementById('loadingOverlay');
        const resultsGrid = document.getElementById('resultsGrid');
        const statusText = document.getElementById('statusText');

        // إدارة أزرار العدد
        document.querySelectorAll('.batch-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                document.querySelectorAll('.batch-btn').forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
                selectedCount = parseInt(btn.getAttribute('data-count'));
            });
        });

        // ضغط الصورة (مهم للسرعة)
        async function compressImage(base64Str) {
            return new Promise((resolve) => {
                const img = new Image();
                img.src = base64Str;
                img.onload = () => {
                    const canvas = document.createElement('canvas');
                    const size = 512;
                    canvas.width = size;
                    canvas.height = size * (img.height / img.width);
                    canvas.getContext('2d').drawImage(img, 0, 0, canvas.width, canvas.height);
                    resolve(canvas.toDataURL('image/jpeg', 0.7).split(',')[1]);
                };
            });
        }

        imageUpload.onchange = async (e) => {
            const file = e.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = async (ev) => {
                    imgPreview.src = ev.target.result;
                    imgPreview.classList.remove('hidden');
                    document.getElementById('uploadPlaceholder').classList.add('hidden');
                    base64Image = await compressImage(ev.target.result);
                };
                reader.readAsDataURL(file);
            }
        };

        analyzeBtn.onclick = async () => {
            if (!base64Image) return;
            loadingOverlay.classList.remove('hidden');
            statusText.textContent = "جاري تحليل ملامح الوجه بدقة...";
            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: "Describe this person's facial features for exact replication. Focus on face shape, eyes, hair. Technical English only." }, { inlineData: { mimeType: "image/jpeg", data: base64Image } }] }]
                    })
                });
                const data = await response.json();
                promptInput.value = `Portrait of: ${data.candidates[0].content.parts[0].text}\n\n[Action: ]`;
            } catch (err) { console.error(err); }
            finally { loadingOverlay.classList.add('hidden'); }
        };

        // وظيفة جلب صورة واحدة (تستخدم للعمل المتوازي)
        async function fetchImage(i) {
            const seed = Math.floor(Math.random() * 10000);
            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image-preview:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: `GENERATE PHOTOREALISTIC IMAGE. Use ref face. Scene: ${promptInput.value}. Seed: ${seed}` }, { inlineData: { mimeType: "image/jpeg", data: base64Image } }] }],
                        generationConfig: { responseModalities: ['TEXT', 'IMAGE'] }
                    })
                });
                const res = await response.json();
                return res.candidates?.[0]?.content?.parts?.find(p => p.inlineData)?.inlineData?.data;
            } catch (e) { return null; }
        }

        // المحرك المتوازي (السرعة القصوى)
        generateBtn.onclick = async () => {
            if (!base64Image || !promptInput.value) return alert("ارفع الصورة وحللها أولاً");
            
            loadingOverlay.classList.remove('hidden');
            statusText.textContent = `جاري توليد ${selectedCount} صور في نفس الوقت...`;
            
            // إنشاء مصفوفة من الوعود (Promises) لتعمل معاً
            const tasks = [];
            for(let i=0; i < selectedCount; i++) {
                tasks.push(fetchImage(i));
            }

            // تنفيذ الكل معاً (Parallel Execution)
            const results = await Promise.all(tasks);

            // عرض النتائج الناجحة
            results.forEach(b64 => {
                if (b64) renderResult(b64);
            });

            loadingOverlay.classList.add('hidden');
        };

        function renderResult(b64) {
            const div = document.createElement('div');
            div.className = "result-card glass-panel p-3 rounded-2xl relative group";
            div.innerHTML = `
                <img src="data:image/jpeg;base64,${b64}" class="w-full h-auto rounded-xl shadow-lg">
                <button class="absolute top-5 right-5 w-10 h-10 bg-black/60 backdrop-blur-md rounded-full text-white flex items-center justify-center opacity-0 group-hover:opacity-100 transition-all hover:bg-blue-600">
                    <i class="fas fa-download text-xs"></i>
                </button>
            `;
            div.querySelector('button').onclick = () => {
                const a = document.createElement('a');
                a.href = `data:image/jpeg;base64,${b64}`;
                a.download = `Result_${Date.now()}.jpg`;
                a.click();
            };
            resultsGrid.prepend(div);
        }
    </script>
